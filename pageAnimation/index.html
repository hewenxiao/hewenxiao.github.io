基于Animation.CSS 和 wow 的页面滚动动画

        第一步引入
        1. animation.css
        2. wow.min.js
    
         第二步引入
        wow = new WOW(
        {
            animateClass: 'animated',
            offset: 200,
            callback: function (box) {
                console.log("WOW: animating <" + box.tagName.toLowerCase() + ">")
            }
        }
    );
    wow.init();
第三步 <div class="wow fadeInLeft dome" data-wow-delay="0.2s">
